def process_file(df):
    df.age = df.age.convert_objects(convert_numeric=True)
    df.gender = df.gender.replace(['?', 'M?'], np.nan)
    
    labels = ['0-50', '51- 99']
    labels = ['0-9', '10-19', '20-29', '30-39', '40-49', '50-59', '60-69', '70-79', '80-89', '90+']
    df['age_group'] = pd.cut(df.age, range(0, 101, 10), right=False, labels=labels)
    
    df['onset'] = df.onset.dropna().apply(convert_date)
    
    return df

def convert_date(x):
    try:
        x = datetime.datetime.strptime(x, '%m/%d/%Y')
    except:
        x = np.nan
        
    return x
